Метод берёт массив и переставляет первый и последний символ местами (если символов 2, то возвращает тот же массив):
class App {
    public static void swap(int[] numbers) {
        var length = numbers.length;

        if (length < 2) {
            return;
        }

        var lastIndex = length - 1;

        var temp = numbers[lastIndex];
        numbers[lastIndex] = numbers[0];
        numbers[0] = temp;
    }
}
